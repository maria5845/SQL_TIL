-- 7-1.WHERE절 배우기 (숫자 데이터 검색) (1) 
    SELECT ename as 이름 , sal as  월급 
        FROM emp

-- ** WHERE절 배우기 (숫자 데이터 검색) (예외)

        FROM emp

    SQL 실행순서 때문에 나는 에러

*/

     SELECT * 
        FROM NLS_SESSION_PARAMETERS

  우리나라의 경우 년도/월/일
